#include <iostream>
using namespace std;

int main() {
    char input[100];
    int i = 0, isNumeric = 1, dotCount = 0;

    cout << "Enter input: ";
    cin >> input;

    // Handle optional sign
    if (input[0] == '+' || input[0] == '-') {
        i = 1;
        if (input[1] == '\0') {
            isNumeric = 0; // only sign, no digits
        }
    }

    while (input[i] != '\0' && isNumeric) {
        if (input[i] == '.') {
            dotCount++;
            if (dotCount > 1) {
                isNumeric = 0; // more than one dot
                break;
            }
        } else if (input[i] < '0' || input[i] > '9') {
            isNumeric = 0; // not a digit
            break;
        }
        i++;
    }

    // Edge case: string is just "." or "+." or "-."
    if (dotCount == 1 && i == 1 + (input[0] == '+' || input[0] == '-')) {
        isNumeric = 0;
    }

    if (isNumeric)
        cout << "This is a Numeric constant." << endl;
    else
        cout << "This is not Numeric." << endl;

    return 0;
}
